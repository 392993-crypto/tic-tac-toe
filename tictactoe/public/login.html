document.addEventListener('DOMContentLoaded', () => { const loginForm = document.getElementById('login-form'); if (loginForm) { loginForm.addEventListener('submit', async (e) => { // Prevent the default form submission behavior (page reload) e.preventDefault(); // Grab the input values const username = document.getElementById('username').value; const password = document.getElementById('password').value; try { // Send a POST request to your backend auth route. // Adjust '/auth/login' if your Express route prefix is different (e.g., '/api/auth/login') const response = await fetch('/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username, password }) }); const data = await response.json(); if (response.ok) { // Optional but recommended: Save the username to localStorage // so your frontend knows who is logged in for the Game and Stats pages localStorage.setItem('currentUser', username); // Redirect to the game page! window.location.href = 'game.html'; } else { // Show whatever error message your Express backend sends (e.g., "User not found") alert(data.message || 'Invalid username or password.'); } } catch (error) { console.error('Login error:', error); alert('A network error occurred. Please try again.'); } }); } });
